import os
from configparser import ConfigParser



[docs]def get_config():
    """
    Returns the config of cqc
    :return: :obj: configparser.ConfigParser
    """
    config = ConfigParser()
    path_to_here = os.path.dirname(os.path.abspath(__file__))
    settings_file = os.path.join(path_to_here, "settings.ini")
    config.read(settings_file)
    return config




[docs]def set_config(config):
    """
    Sets the config of cqc
    :param config: :obj: configparser.ConfigParser
    :return: None
    """
    path_to_here = os.path.dirname(os.path.abspath(__file__))
    settings_file = os.path.join(path_to_here, "settings.ini")
    with open(settings_file, 'w') as f:
        config.write(f)



[docs]def get_cqc_file():
    """
    Gets the path to the CQC file used
    :return: str
    """
    config = get_config()
    try:
        file_path = config['FILEPATHS']['cqc_file']
    except KeyError:
        return None

    return file_path
